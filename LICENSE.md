## FizzBuzz
```java
const var i: Int!
when (i % 3 = 0 && i % 5 = 0) print("FizzBuzz")!
when (i % 3 = 0 && i % 5 != 0) print("Fizz")!
when (i % 3 != 0 && i % 5 = 0) print("Buzz")!
when (i % 3 != 0 && i % 5 != 0) print(i)!

when (i < 20) i++!
i = 0!
```
